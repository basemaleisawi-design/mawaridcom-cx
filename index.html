var CONFIG = {
  SHEET_ID: 'PASTE_YOUR_SHEET_ID_HERE',
  MANAGER_EMAIL: 'basemaleisawi@gmail.com',
  NPS_THRESHOLD: 7,
  SHEETS: {
    companies:  'شركات',
    candidates: 'مرشحون',
    outsource:  'موظفون خارجيون'
  }
};

var HEADERS = {
  companies:  ['التاريخ','الاسم','البريد','الجوال','الشركة','CSAT','NPS','CES','التعليق','اللغة'],
  candidates: ['التاريخ','الاسم','البريد','الجوال','CSAT','NPS','CES','اختيارات','التعليق','اللغة'],
  outsource:  ['التاريخ','الاسم','البريد','الجوال','الشركة المضيفة','CSAT','NPS','CES','مشاكل راتب','التعليق','اللغة']
};

function getSheet(ss, type) {
  var name = CONFIG.SHEETS[type] || 'عام';
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    var hdrs = HEADERS[type] || ['التاريخ','البيانات'];
    sheet.appendRow(hdrs);
    sheet.getRange(1,1,1,hdrs.length)
      .setBackground('#1b347e')
      .setFontColor('#ffffff')
      .setFontWeight('bold');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.openById(CONFIG.SHEET_ID);
    var sheet = getSheet(ss, d.type);
    var now = new Date().toLocaleString('ar-SA',{timeZone:'Asia/Riyadh'});
    var row;
    if (d.type === 'companies') {
      row = [now,d.name||'',d.email||'',d.phone||'',d.company||'',d.csat||'',d.nps||'',d.ces||'',d.comment||'',d.lang||'ar'];
    } else if (d.type === 'candidates') {
      row = [now,d.name||'',d.email||'',d.phone||'',d.csat||'',d.nps||'',d.ces||'',(d.choices||[]).join(', '),d.comment||'',d.lang||'ar'];
    } else {
      row = [now,d.name||'',d.email||'',d.phone||'',d.company||'',d.csat||'',d.nps||'',d.ces||'',d.salaryIssue||'',d.comment||'',d.lang||'ar'];
    }
    sheet.appendRow(row);
    if (d.nps !== undefined && parseInt(d.nps) < CONFIG.NPS_THRESHOLD) {
      var sub = '🚨 تنبيه مواردكم: NPS منخفض - ' + (d.name||'عميل');
      var body = 'الاسم: '+(d.name||'-')+'\nNPS: '+d.nps+'/10\nCSAT: '+(d.csat||'-')+'/5\nتعليق: '+(d.comment||'-');
      GmailApp.sendEmail(CONFIG.MANAGER_EMAIL, sub, body);
    }
    return ContentService
      .createTextOutput(JSON.stringify({success:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({success:false,error:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok',system:'MawaridcomCX v3'}))
    .setMimeType(ContentService.MimeType.JSON);
}
